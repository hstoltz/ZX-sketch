% learn-rules.tex — ZX-calculus rule diagrams for the Learn More overlay.

% Page index (LHS=odd, RHS=even):
%  1-2:   Spider Fusion         15-16: Local Complementation
%  3-4:   Identity Removal      17-18: Pivoting
%  5-6:   Bialgebra             19-20: Boundary Pivot
%  7-8:   Hopf Rule             21-22: Gadget Pivot
%  9-10:  Copy Rule             23-24: Phase Gadget Fusion

%  31-32: Euler Decomposition

%  35-36: Wire Vertex
\documentclass[tikz,border=6pt,multi=true]{standalone}
\usepackage{zx-calculus}
\usepackage{amsmath}
\standaloneenv{ZX}
% Phaseless spiders get [minimum size=4mm] inline to match phased spider size.
% Package default for phaseless is 2mm, phased is 5mm at 0.8x scale ≈ 4mm.
\begin{document}

% 1. Spider Fusion — LHS
\begin{ZX}
  \zxN{} \ar[r] & \zxZ{\alpha} \ar[r] & \zxZ{\beta} \ar[r] & \zxN{}
\end{ZX}

% 2. Spider Fusion — RHS

% 3. Identity Removal — LHS

\end{ZX}

% 4. Identity Removal — RHS
\begin{ZX}

\end{ZX}

% 5. Bialgebra — LHS
\begin{ZX}
  \zxN{} \ar[rd] & & & \zxN{} \\
  & \zxZ[minimum size=4mm]{} \ar[r] & \zxX[minimum size=4mm]{} \ar[ru] \ar[rd] & \\
  \zxN{} \ar[ru] & & & \zxN{}
\end{ZX}

% 6. Bialgebra — RHS
\begin{ZX}
  \zxN{} \ar[r] & \zxX[minimum size=4mm]{} \ar[r] \ar[rd] & \zxZ[minimum size=4mm]{} \ar[r] & \zxN{} \\
  \zxN{} \ar[r] & \zxX[minimum size=4mm]{} \ar[r] \ar[ru] & \zxZ[minimum size=4mm]{} \ar[r] & \zxN{}
\end{ZX}

% 7. Hopf Rule — LHS
\begin{ZX}
  \zxN{} \ar[r] & \zxZ[minimum size=4mm]{} \ar[r,H,bend left=30] \ar[r,H,bend right=30] &[5mm] \zxX[minimum size=4mm]{} \ar[r] & \zxN{}
\end{ZX}

% 8. Hopf Rule — RHS
\begin{ZX}
  \zxN{} \ar[r] & \zxZ[minimum size=4mm]{} & & \zxX[minimum size=4mm]{} \ar[r] & \zxN{}
\end{ZX}

% 9. Copy Rule — LHS
\begin{ZX}
  \zxN{} \ar[rd] & &[4mm] \\
  & \zxX[minimum size=4mm]{} \ar[r,H] & \zxZ[minimum size=4mm]{} \\
  \zxN{} \ar[ru] & &
\end{ZX}

% 10. Copy Rule — RHS
\begin{ZX}
  \zxN{} \ar[r] & \zxZ[minimum size=4mm]{} \\
  \zxN{} \ar[r] & \zxZ[minimum size=4mm]{}
\end{ZX}

% 11. Color Change — LHS
\begin{ZX}
  \zxN{} \ar[r] & \zxZ{\alpha} \ar[r] & \zxN{}
\end{ZX}

% 12. Color Change — RHS
\begin{ZX}
  \zxN{} \ar[r,H] &[4mm] \zxX{\alpha} \ar[r,H] &[4mm] \zxN{}
\end{ZX}

% 13. Self-Loop Removal — LHS
\begin{ZX}
  \zxN{} \ar[r] & \zxZ[a]{\alpha} \ar[r] \ar[out=110,in=70,loop] & \zxN{}
\end{ZX}

% 14. Self-Loop Removal — RHS
\begin{ZX}
  \zxN{} \ar[r] & \zxZ{\alpha} \ar[r] & \zxN{}
\end{ZX}

% 15. Local Complementation — LHS
\begin{ZX}
  & \zxZ[minimum size=4mm]{} \ar[d,H] & \\
  \zxZ[minimum size=4mm]{} \ar[r,H] & \zxZ{\pm\frac{\pi}{2}} \ar[r,H] & \zxZ[minimum size=4mm]{} \\
  & \zxZ[minimum size=4mm]{} \ar[u,H] &
\end{ZX}

% 16. Local Complementation — RHS
\begin{ZX}
  \zxZ[minimum size=4mm]{} \ar[r,H] \ar[rd,H] & \zxZ[minimum size=4mm]{} \ar[d,H] \\
  \zxZ[minimum size=4mm]{} \ar[u,H] \ar[r,H] & \zxZ[minimum size=4mm]{}
\end{ZX}

% 17. Pivoting — LHS
\begin{ZX}
  \zxZ[minimum size=4mm]{} \ar[rd,H] & & \zxZ[minimum size=4mm]{} \\
  & \zxZ[minimum size=4mm]{} \ar[r,H] & \zxZ[minimum size=4mm]{} \ar[u,H] \ar[d,H] \\
  \zxZ[minimum size=4mm]{} \ar[ru,H] & & \zxZ[minimum size=4mm]{}
\end{ZX}

% 18. Pivoting — RHS

  \zxZ[minimum size=4mm]{} \ar[r,H] \ar[ru,H] & \zxZ[minimum size=4mm]{}
\end{ZX}

% 19. Boundary Pivot — LHS
\begin{ZX}
  & & \zxZ[minimum size=4mm]{} \\
  \zxN{} \ar[r] & \zxZ[minimum size=4mm]{} \ar[r,H] & \zxZ[minimum size=4mm]{} \ar[u,H] \ar[d,H] \\
  & & \zxZ[minimum size=4mm]{}
\end{ZX}

% 20. Boundary Pivot — RHS
\begin{ZX}
  & \zxZ[minimum size=4mm]{} \\
  \zxN{} \ar[r] & \zxZ[minimum size=4mm]{} \ar[u,H] \ar[d,H] \\
  & \zxZ[minimum size=4mm]{}

% 21. Gadget Pivot — LHS
\begin{ZX}
  & & & \zxZ[minimum size=4mm]{} \\

% 22. Gadget Pivot — RHS

% 23. Phase Gadget Fusion — LHS

% 24. Phase Gadget Fusion — RHS
\begin{ZX}

  & \zxZ[minimum size=4mm]{} \ar[r] & \zxZ{\alpha\!+\!\beta} \\

\end{ZX}

% 25. Supplementarity — LHS

% 26. Supplementarity — RHS

  \zxZ[minimum size=4mm]{} \ar[r,H] & \zxZ[minimum size=4mm]{}
\end{ZX}

% 27. Unfuse — LHS
